% ViT3D depth-6 encoder — emphasises spatiotemporal tokenisation.
% Stops at the token cube + cross-time attention message. The downstream
% Tx blocks / reshape / mean steps are described in prose, not drawn.
\begin{tikzpicture}[
  node distance = 7mm and 9mm,
  every node/.style = {font=\scriptsize, inner sep=1.5pt}]

  % =====================================================================
  % 1. Input frame stack — 8 isometric slabs, front 4 highlighted
  % =====================================================================
  \begin{scope}[local bounding box=stack]
    \foreach \i [count=\k from 0] in {7,6,5,4,3,2,1,0} {
      \pgfmathsetmacro{\xs}{0.18*\i}
      \pgfmathsetmacro{\ys}{-0.18*\i}
      \pgfmathsetmacro{\op}{20+9*\k}
      \ifnum\i<4
        \draw[fill=red!\op, draw=red!70!black, thick]
          (\xs, \ys) rectangle ++(1.0, 1.0);
        \draw[fill=red!75, draw=red!90!black, thick]
          (\xs+0.10, \ys+0.55) rectangle ++(0.32, 0.32);
      \else
        \draw[fill=blue!\op, draw=blue!60!black, thick]
          (\xs, \ys) rectangle ++(1.0, 1.0);
      \fi
    }
  \end{scope}
  \draw[->, thick, gray!70]
    ($(stack.south west) + (-0.05, -0.08)$) --
    ($(stack.south east) + (0.05, -0.08)$)
    node[midway, below=1pt, font=\tiny, gray!80]
    {time $\rightarrow$ (16 frames)};
  \node[red!70!black, font=\tiny, align=center, anchor=south east]
    at ($(stack.north west) + (-0.0, 0.05)$)
    {1 patch =\\4 frames\\$\times\,16{\times}16$};
  \node[meta, above=12mm of stack.center]
    {$(B,\mathbf{11},16,256,256)$};

  % =====================================================================
  % 2. PatchEmbed conv
  % =====================================================================
  \node[op, right=12mm of stack.east, align=center, minimum width=24mm,
        font=\tiny]
    (pe) {3D PatchEmbed\\Conv3d $4{\times}16{\times}16$\\$11\!\to\!384$};
  \draw[arrow, thick] (stack.east) -- (pe.west);

  % =====================================================================
  % 3. Dense 5-D feature map (B, 384, 4, 16, 16)
  % =====================================================================
  \node[right=8mm of pe.east] (fmap_anchor) {};
  \begin{scope}[local bounding box=fmap]
    \foreach \i [count=\k from 0] in {3,2,1,0} {
      \pgfmathsetmacro{\xs}{0.20*\i + 0.0}
      \pgfmathsetmacro{\ys}{-0.20*\i + 0.0}
      \pgfmathsetmacro{\op}{30+15*\k}
      \draw[fill=red!\op, draw=red!75!black, thick]
        ($(fmap_anchor) + (\xs, \ys)$) rectangle ++(0.85, 0.85);
    }
  \end{scope}
  \draw[arrow, thick] (pe.east) -- ($(fmap.west) + (0,0.1)$);
  \node[font=\tiny, gray!85, align=center, below=2mm of fmap]
    {$(B,\mathbf{384},4,16,16)$};

  % =====================================================================
  % 4. flatten + transpose -> token cube (B, 1024, 384)
  % =====================================================================
  \node[op, right=8mm of fmap.east, align=center, font=\tiny,
        minimum width=18mm, fill=yellow!18]
    (rs) {flatten $T'H'W'$\\$+$ transpose};
  \draw[arrow, thick] (fmap.east) -- (rs.west);

  \node[right=10mm of rs.east] (cube_anchor) {};
  \begin{scope}[local bounding box=cube]
    \foreach \i [count=\k from 0] in {3,2,1,0} {
      \pgfmathsetmacro{\xs}{0.42*\i}
      \pgfmathsetmacro{\ys}{-0.42*\i}
      \pgfmathsetmacro{\op}{30+18*\k}
      \draw[fill=red!\op, draw=red!75!black, thick]
        ($(cube_anchor) + (\xs, \ys)$) rectangle ++(1.2, 1.2);
      \foreach \g in {0.30, 0.60, 0.90} {
        \draw[red!25, very thin]
          ($(cube_anchor) + (\xs+\g, \ys)$) -- ++(0, 1.2);
        \draw[red!25, very thin]
          ($(cube_anchor) + (\xs, \ys+\g)$) -- ++(1.2, 0);
      }
      \node[font=\tiny, red!90!black, anchor=north east]
        at ($(cube_anchor) + (\xs+1.2, \ys)$) {$t'{=}\k$};
    }
    % cross-time attention edges
    \draw[<->, very thick, black]
      ($(cube_anchor) + (0.45, 0.90)$)
      .. controls ($(cube_anchor) + (0.55, 1.65)$) and
                  ($(cube_anchor) + (1.55, 1.65)$) ..
      ($(cube_anchor) + (0.45+0.42*3, 0.90-0.42*3)$);
    \draw[<->, very thick, black]
      ($(cube_anchor) + (0.20, 0.45)$)
      .. controls ($(cube_anchor) + (0.30, 1.30)$) and
                  ($(cube_anchor) + (1.40, 1.20)$) ..
      ($(cube_anchor) + (0.20+0.72*3, 1.25-0.72*3)$);
    \draw[<->, very thick, black]
      ($(cube_anchor) + (0.85, 0.30)$) --
      ($(cube_anchor) + (0.85+0.42*3, 0.30-0.42*3)$);
  \end{scope}
  \draw[arrow, thick] (rs.east) -- ($(cube.west) + (0,0.15)$)
    node[midway, above, font=\tiny, gray!85]{$+$\,PE};

  \node[font=\tiny, black, align=center, anchor=south]
    at ($(cube.north) + (0, 0.20)$)
    {self-attention mixes\\across time \& space};
  \node[font=\tiny, gray!85, align=center, below=2mm of cube]
    {$(B,\mathbf{1024},384)$};

\end{tikzpicture}
